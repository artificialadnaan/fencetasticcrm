```typescript
import { Prisma } from '@prisma/client';
import { prisma } from '../lib/prisma';
import { AppError } from '../middleware/error-handler';
import type {
  TransactionListQuery,
  TransactionSummary,
  CreateTransactionDTO,
  UpdateTransactionDTO,
} from '@fencetastic/shared';

function d(val: Prisma.Decimal | null | undefined): number {
  if (!val) return 0;
  if (typeof (val as unknown as { toNumber?: () => number }).toNumber === 'function') {
    return (val as unknown as { toNumber: () => number }).toNumber();
  }
  return Number(val);
}

export async function listTransactions(query: TransactionListQuery = {}) {
  const { page = 1, limit = 20, type, category, projectId, dateFrom, dateTo } = query;

  const where: Prisma.TransactionWhereInput = {};
  if (type) where.type = type;
  if (category) where.category = category;
  if (projectId) where.projectId = projectId;
  if (dateFrom || dateTo) {
    where.date = {};
    if (dateFrom) where.date.gte = new Date(dateFrom);
    if (dateTo) where.date.lte = new Date(dateTo);
  }

  const [transactions, total] = await Promise.all([
    prisma.transaction.findMany({
      where,
      include: { project: { select: { id: true, customer: true } } },
      orderBy: { date: 'desc' },
      skip: (page - 1) * limit,
      take: limit,
    }),
    prisma.transaction.count({ where }),
  ]);

  const data = transactions.map((t) => ({
    id: t.id,
    type: t.type,
    amount: d(t.amount),
    date: t.date.toISOString(),
    category: t.category,
    description: t.description,
    payee: t.payee,
    paymentMethod: t.paymentMethod,
    projectId: t.projectId,
    projectCustomer: t.project?.customer ?? null,
    sourceField: t.sourceField,
    isAutoGenerated: t.isAutoGenerated,
    createdAt: t.createdAt.toISOString(),
    updatedAt: t.updatedAt.toISOString(),
  }));

  return {
    data,
    pagination: { page, limit, total, totalPages: Math.ceil(total / limit) },
  };
}

export async function getTransactionSummary(period: 'mtd' | 'ytd') {
  const now = new Date();
  const startDate =
    period === 'mtd'
      ? new Date(now.getFullYear(), now.getMonth(), 1)
      : new Date(now.getFullYear(), 0, 1);

  const transactions = await prisma.transaction.findMany({
    where: { date: { gte: startDate, lte: now } },
    select: { type: true, amount: true },
  });

  let totalIncome = 0;
  let totalExpenses = 0;
  for (const t of transactions) {
    const amt = d(t.amount);
    if (t.type === 'INCOME') totalIncome += amt;
    else totalExpenses += Math.abs(amt);
  }

  const summary: TransactionSummary = {
    totalIncome: Number(totalIncome.toFixed(2)),
    totalExpenses: Number(totalExpenses.toFixed(2)),
    netProfit: Number((totalIncome - totalExpenses).toFixed(2)),
  };
  return summary;
}

export async function createTransaction(dto: CreateTransactionDTO) {
  const transaction = await prisma.transaction.create({
    data: {
      type: dto.type,
      amount: dto.amount,
      date: new Date(dto.date),
      category: dto.category,
      description: dto.description,
      payee: dto.payee ?? null,
      paymentMethod: dto.paymentMethod ?? null,
      projectId: dto.projectId ?? null,
      isAutoGenerated: false,
    },
  });
  return { id: transaction.id };
}

export async function updateTransaction(id: string, dto: UpdateTransactionDTO) {
  const existing = await prisma.transaction.findUnique({ where: { id } });
  if (!existing) throw new AppError(404, 'Transaction not found', 'TRANSACTION_NOT_FOUND');

  const updateData: Prisma.TransactionUpdateInput = {};
  if (dto.type !== undefined) updateData.type = dto.type;
  if (dto.amount !== undefined) updateData.amount = dto.amount;
  if (dto.date !== undefined) updateData.date = new Date(dto.date);
  if (dto.category !== undefined) updateData.category = dto.category;
  if (dto.description !== undefined) updateData.description = dto.description;
  if (dto.payee !== undefined) updateData.payee = dto.payee;
  if (dto.paymentMethod !== undefined) updateData.paymentMethod = dto.paymentMethod;
  if (dto.projectId !== undefined) {
    updateData.project = dto.projectId ? { connect: { id: dto.projectId } } : { disconnect: true };
  }

  await prisma.transaction.update({ where: { id }, data: updateData });
  return { id };
}

export async function deleteTransaction(id: string) {
  const existing = await prisma.transaction.findUnique({ where: { id } });
  if (!existing) throw new AppError(404, 'Transaction not found', 'TRANSACTION_NOT_FOUND');
  await prisma.transaction.delete({ where: { id } });
}

// Called from project.service.ts when tracked fields change
export async function createAutoTransaction(
  projectId: string,
  type: 'INCOME' | 'EXPENSE',
  category: string,
  sourceField: string,
  delta: number,
  payee: string
) {
  if (delta === 0) return;
  await prisma.transaction.create({
    data: {
      type,
      amount: delta,
      date: new Date(),
      category,
      description: `Auto: ${category} for ${payee}`,
      payee,
      projectId,
      sourceField,
      isAutoGenerated: true,
    },
  });
}

export async function getMonthlyBreakdown() {
  const transactions = await prisma.transaction.findMany({
    select: { type: true, amount: true, date: true },
    orderBy: { date: 'asc' },
  });

  const months: Record<string, { income: number; expenses: number }> = {};
  for (const t of transactions) {
    const key = `${t.date.getFullYear()}-${String(t.date.getMonth() + 1).padStart(2, '0')}`;
    if (!months[key]) months[key] = { income: 0, expenses: 0 };
    const amt = d(t.amount);
    if (t.type === 'INCOME') months[key].income += amt;
    else months[key].expenses += Math.abs(amt);
  }

  return Object.entries(months)
    .sort(([a], [b]) => a.localeCompare(b))
    .map(([month, data]) => ({
      month,
      income: Number(data.income.toFixed(2)),
      expenses: Number(data.expenses.toFixed(2)),
    }));
}

export async function getCategoryBreakdown() {
  const transactions = await prisma.transaction.findMany({
    where: { type: 'EXPENSE' },
    select: { category: true, amount: true },
  });

  const cats: Record<string, number> = {};
  for (const t of transactions) {
    const amt = Math.abs(d(t.amount));
    cats[t.category] = (cats[t.category] || 0) + amt;
  }

  return Object.entries(cats).map(([category, total]) => ({
    category,
    total: Number(total.toFixed(2)),
  }));
}
```

```typescript
export async function listProjectsGrid(query: ProjectListQuery = {}) {
  const {
    page = 1,
    limit = 50,
    sortBy = 'installDate',
    sortDir = 'desc',
    status,
    fenceType,
    search,
    dateFrom,
    dateTo,
  } = query;

  const where: Prisma.ProjectWhereInput = { isDeleted: false };
  if (status) where.status = status;
  if (fenceType) where.fenceType = fenceType;
  if (search) {
    where.OR = [
      { customer: { contains: search, mode: 'insensitive' } },
      { address: { contains: search, mode: 'insensitive' } },
      { subcontractor: { contains: search, mode: 'insensitive' } },
    ];
  }
  if (dateFrom || dateTo) {
    where.installDate = {};
    if (dateFrom) where.installDate.gte = new Date(dateFrom);
    if (dateTo) where.installDate.lte = new Date(dateTo);
  }

  const sortFieldMap: Record<string, string> = {
    customer: 'customer', address: 'address', fenceType: 'fenceType',
    status: 'status', projectTotal: 'projectTotal', installDate: 'installDate',
    contractDate: 'contractDate', createdAt: 'createdAt',
  };
  const orderField = sortFieldMap[sortBy] || 'installDate';

  const [projects, total] = await Promise.all([
    prisma.project.findMany({
      where,
      include: {
        subcontractorPayments: {
          select: { amountPaid: true },
          orderBy: { id: 'asc' },
          take: 2,
        },
      },
      orderBy: { [orderField]: sortDir },
      skip: (page - 1) * limit,
      take: limit,
    }),
    prisma.project.count({ where }),
  ]);

  const data = projects.map((p) => {
    const projectTotal = d(p.projectTotal);
    const moneyReceived = d(p.moneyReceived);
    const customerPaid = d(p.customerPaid);
    const forecastedExpenses = d(p.forecastedExpenses);
    const materialsCost = d(p.materialsCost);
    const commissionOwed = d(p.commissionOwed);
    const commissionPaid = d(p.commissionPaid);
    const memesCommission = d(p.memesCommission);
    const aimannsCommission = d(p.aimannsCommission);

    const outstandingReceivables = Number((projectTotal - customerPaid).toFixed(2));
    const outstandingPayables = Number((commissionOwed - commissionPaid).toFixed(2));
    const profitDollar = Number((moneyReceived - forecastedExpenses - commissionOwed).toFixed(2));
    const profitPercent = projectTotal > 0 ? Number(((profitDollar / projectTotal) * 100).toFixed(2)) : 0;
    const netProfitDollar = Number((profitDollar - memesCommission - aimannsCommission).toFixed(2));
    const netProfitPercent = projectTotal > 0 ? Number(((netProfitDollar / projectTotal) * 100).toFixed(2)) : 0;

    const subPayments = p.subcontractorPayments;

    return {
      id: p.id,
      installDate: p.installDate?.toISOString().split('T')[0] ?? null,
      status: p.status,
      contractDate: p.contractDate.toISOString().split('T')[0],
      notes: p.notes,
      subcontractor: p.subcontractor,
      customer: p.customer,
      address: p.address,
      description: p.description,
      fenceType: p.fenceType,
      projectTotal,
      moneyReceived,
      customerPaid,
      paymentMethod: p.paymentMethod,
      outstandingReceivables,
      forecastedExpenses,
      materialsCost,
      subPayment1: subPayments[0] ? d(subPayments[0].amountPaid) : null,
      subPayment2: subPayments[1] ? d(subPayments[1].amountPaid) : null,
      commissionOwed,
      commissionPaid,
      outstandingPayables,
      profitDollar,
      profitPercent,
      memesCommission,
      aimannsCommission,
      netProfitDollar,
      netProfitPercent,
    };
  });

  return {
    data,
    pagination: { page, limit, total, totalPages: Math.ceil(total / limit) },
  };
}
```

```typescript
function d(val: Prisma.Decimal | null | undefined): number {
  if (!val) return 0;
  if (typeof (val as unknown as { toNumber?: () => number }).toNumber === 'function') {
    return (val as unknown as { toNumber: () => number }).toNumber();
  }
  return Number(val);
}

export async function getWorkOrder(projectId: string) {
  const wo = await prisma.workOrder.findFirst({
    where: { projectId },
    include: { segments: { orderBy: { segmentNumber: 'asc' } } },
  });
  if (!wo) return null;

  return {
    id: wo.id,
    projectId: wo.projectId,
    drawingData: wo.drawingData,
    propertyNotes: wo.propertyNotes,
    segments: wo.segments.map((s) => ({

      style: s.style,
      height: d(s.height),
      linearFeet: d(s.linearFeet),

}

export async function createWorkOrder(projectId: string, dto: CreateWorkOrderDTO) {
  const project = await prisma.project.findUnique({ where: { id: projectId } });
  if (!project || project.isDeleted) {
    throw new AppError(404, 'Project not found', 'PROJECT_NOT_FOUND');
  }

  const existing = await prisma.workOrder.findFirst({ where: { projectId } });
  if (existing) {
    throw new AppError(400, 'Work order already exists for this project', 'WORK_ORDER_EXISTS');
  }

  // Validate drawingData size (5MB max)
  const jsonSize = JSON.stringify(dto.drawingData).length;
  if (jsonSize > 5 * 1024 * 1024) {
    throw new AppError(400, 'Drawing data exceeds 5MB limit', 'DRAWING_TOO_LARGE');
  }

  const wo = await prisma.workOrder.create({
    data: {
      projectId,
      drawingData: dto.drawingData as Prisma.InputJsonValue,
      propertyNotes: dto.propertyNotes ?? null,
      segments: {
        create: dto.segments.map((s, i) => ({
          segmentNumber: s.segmentNumber ?? i + 1,
          fenceType: s.fenceType,
          style: s.style,
          height: s.height,
          linearFeet: s.linearFeet,
          steps: s.steps as Prisma.InputJsonValue ?? Prisma.JsonNull,
          additions: s.additions as unknown as Prisma.InputJsonValue,
          customAdditions: s.customAdditions as unknown as Prisma.InputJsonValue,
          notes: s.notes ?? null,
        })),
      },
    },
  });

  return { id: wo.id };
}

export async function updateWorkOrder(workOrderId: string, dto: UpdateWorkOrderDTO) {
  const wo = await prisma.workOrder.findUnique({ where: { id: workOrderId } });
  if (!wo) throw new AppError(404, 'Work order not found', 'WORK_ORDER_NOT_FOUND');

  if (dto.drawingData) {
    const jsonSize = JSON.stringify(dto.drawingData).length;
    if (jsonSize > 5 * 1024 * 1024) {
      throw new AppError(400, 'Drawing data exceeds 5MB limit', 'DRAWING_TOO_LARGE');
    }
  }

  await prisma.$transaction(async (tx) => {
    const updateData: Prisma.WorkOrderUpdateInput = {};
    if (dto.drawingData !== undefined) updateData.drawingData = dto.drawingData as Prisma.InputJsonValue;
    if (dto.propertyNotes !== undefined) updateData.propertyNotes = dto.propertyNotes;

    await tx.workOrder.update({ where: { id: workOrderId }, data: updateData });

    if (dto.segments !== undefined) {
      await tx.fenceSegment.deleteMany({ where: { workOrderId } });
      if (dto.segments.length > 0) {
        await tx.fenceSegment.createMany({
          data: dto.segments.map((s, i) => ({
            workOrderId,
            segmentNumber: s.segmentNumber ?? i + 1,
            fenceType: s.fenceType,
            style: s.style,
            height: s.height,
            linearFeet: s.linearFeet,
            steps: s.steps as Prisma.InputJsonValue ?? Prisma.JsonNull,
            additions: s.additions as unknown as Prisma.InputJsonValue,
            customAdditions: s.customAdditions as unknown as Prisma.InputJsonValue,
            notes: s.notes ?? null,
          })),
        });
      }
    }
  });

  return { id: workOrderId };
}

export async function deleteWorkOrder(workOrderId: string) {
  const wo = await prisma.workOrder.findUnique({ where: { id: workOrderId } });
  if (!wo) throw new AppError(404, 'Work order not found', 'WORK_ORDER_NOT_FOUND');
  await prisma.workOrder.delete({ where: { id: workOrderId } });
}

export async function getWorkOrderForPdf(workOrderId: string) {

    where: { id: workOrderId },
    include: {
      project: { select: { customer: true, address: true, description: true } },
      segments: { orderBy: { segmentNumber: 'asc' } },
    },
  });
  if (!wo) throw new AppError(404, 'Work order not found', 'WORK_ORDER_NOT_FOUND');
  return wo;
}
```

```typescript
import { useState, useEffect, useCallback } from 'react';
import { api } from '@/lib/api';
import type { GridProjectRow, ProjectListQuery } from '@fencetastic/shared';

export function useGridProjects(query: ProjectListQuery = {}) {
  const [data, setData] = useState<GridProjectRow[]>([]);
  const [pagination, setPagination] = useState<{ page: number; limit: number; total: number; totalPages: number } | null>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const queryKey = JSON.stringify(query);

  const fetchData = useCallback(async () => {
    try {
      setError(null);
      const params = new URLSearchParams();
      const q: ProjectListQuery = JSON.parse(queryKey);
      if (q.page) params.set('page', String(q.page));
      if (q.limit) params.set('limit', String(q.limit));
      if (q.sortBy) params.set('sortBy', q.sortBy);
      if (q.sortDir) params.set('sortDir', q.sortDir);
      if (q.status) params.set('status', q.status);
      if (q.fenceType) params.set('fenceType', q.fenceType);
      if (q.search) params.set('search', q.search);
      if (q.dateFrom) params.set('dateFrom', q.dateFrom);
      if (q.dateTo) params.set('dateTo', q.dateTo);
      const res = await api.get(`/projects/grid?${params.toString()}`);
      setData(res.data.data);
      setPagination(res.data.pagination);
    } catch (err: unknown) {
      setError(err instanceof Error ? err.message : 'Failed to load projects');
    } finally {
      setIsLoading(false);
    }
  }, [queryKey]);

  useEffect(() => { setIsLoading(true); fetchData(); }, [fetchData]);

  return { data, pagination, isLoading, error, refetch: fetchData };
}
```

```typescript
              activeTab === tab.value ? 'border-brand-purple text-brand-purple' : 'border-transparent text-gray-500 hover:text-gray-700'
            }`}
          >
            {tab.label}
          </button>

      </div>

      {/* Search */}
      <div className="relative">
        <Search className="absolute left-3 top-2.5 h-4 w-4 text-gray-400" />

          className="w-full pl-9 pr-3 py-2 border rounded-lg text-sm"
          value={search}
          onChange={(e) => { setSearch(e.target.value); setPage(1); }}
        />
      </div>

      {/* Grid */}
      {isLoading ? (
        <div className="flex justify-center py-12">
          <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-brand-purple" />
        </div>
      ) : (
        <div className="border rounded-lg overflow-auto max-h-[calc(100vh-280px)]">
          <table className="w-full text-sm">
            <thead className="sticky top-0 bg-gray-50 z-10">
              {table.getHeaderGroups().map((hg) => (
                <tr key={hg.id}>
                  {hg.headers.map((header) => (
                    <th key={header.id} className="px-2 py-2 text-left font-medium text-gray-600 whitespace-nowrap border-b" style={{ minWidth: header.getSize() }}>
                      {flexRender(header.column.columnDef.header, header.getContext())}

              ))}
            </thead>
            <tbody>
              {table.getRowModel().rows.map((row) => (
                <tr key={row.id} className={`border-b hover:bg-gray-50 ${STATUS_COLORS[row.original.status] || ''}`}>
                  {row.getVisibleCells().map((cell) => (
                    <td key={cell.id} className="px-2 py-1.5 whitespace-nowrap">
                      {flexRender(cell.column.columnDef.cell, cell.getContext())}
                    </td>
                  ))}
                </tr>
              ))}
              {/* Totals row */}
              <tr className="bg-gray-100 font-bold border-t-2">
                <td className="px-2 py-2" colSpan={9}>TOTALS</td>
                <td className="px-2 py-2">{formatCurrency(totals.projectTotal)}</td>
                <td className="px-2 py-2">{formatCurrency(totals.moneyReceived)}</td>
                <td className="px-2 py-2">{formatCurrency(totals.customerPaid)}</td>
                <td className="px-2 py-2"></td>
                <td className="px-2 py-2">{formatCurrency(totals.outstandingReceivables)}</td>
                <td className="px-2 py-2">{formatCurrency(totals.forecastedExpenses)}</td>
                <td className="px-2 py-2">{formatCurrency(totals.materialsCost)}</td>
                <td className="px-2 py-2" colSpan={5}></td>
                <td className="px-2 py-2">{formatCurrency(totals.profitDollar)}</td>
                <td className="px-2 py-2" colSpan={4}></td>
                <td className="px-2 py-2">{formatCurrency(totals.netProfitDollar)}</td>
                <td className="px-2 py-2"></td>
              </tr>
            </tbody>
          </table>
        </div>
      )}

      {/* Pagination */}
      {pagination && pagination.totalPages > 1 && (
        <div className="flex justify-center gap-2">
          <Button variant="outline" size="sm" disabled={page <= 1} onClick={() => setPage(page - 1)}>Previous</Button>
          <span className="px-3 py-1 text-sm">Page {page} of {pagination.totalPages}</span>
          <Button variant="outline" size="sm" disabled={page >= pagination.totalPages} onClick={() => setPage(page + 1)}>Next</Button>
        </div>
      )}
    </div>
  );
}
```

```typescript
import { useState, useEffect, useCallback } from 'react';
import { api } from '@/lib/api';
import type { WorkOrderData, CreateWorkOrderDTO, UpdateWorkOrderDTO } from '@fencetastic/shared';

export function useWorkOrder(projectId: string) {
  const [workOrder, setWorkOrder] = useState<WorkOrderData | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  const fetch = useCallback(async () => {
    try {
      const res = await api.get(`/projects/${projectId}/work-order`);
      setWorkOrder(res.data.data);
    } catch {
      setWorkOrder(null);
    } finally {
      setIsLoading(false);
    }
  }, [projectId]);

  useEffect(() => { fetch(); }, [fetch]);

  const create = async (dto: CreateWorkOrderDTO) => {
    const res = await api.post(`/projects/${projectId}/work-order`, dto);
    await fetch();
    return res.data.data;
  };

  const update = async (workOrderId: string, dto: UpdateWorkOrderDTO) => {
    await api.patch(`/work-orders/${workOrderId}`, dto);
    await fetch();
  };

  const generatePdf = async (workOrderId: string, drawingImage: string) => {
    const res = await api.post(`/work-orders/${workOrderId}/pdf`, { drawingImage }, { responseType: 'blob' });
    const url = URL.createObjectURL(res.data);
    const a = document.createElement('a');
    a.href = url;
    a.download = `work-order-${workOrderId}.pdf`;
    a.click();
    URL.revokeObjectURL(url);
  };

  return { workOrder, isLoading, create, update, generatePdf, refetch: fetch };
}
```

```typescript

  Undo2, Redo2, Save, FileDown, ArrowLeft, Plus,

interface CanvasElement {
  id: string;
  type: 'house' | 'fence' | 'gate' | 'property' | 'text' | 'measure';
  x: number;
  y: number;
  width?: number;
  height?: number;
  points?: number[];
  text?: string;
  rotation?: number;
}

const GRID_SIZE = 20; // pixels per foot
const FENCE_TYPES = ['Wood B/B', 'Wood S/S', 'Wood Raw', 'Iron', 'Chain Link', 'Vinyl'];

export default function WorkOrderPage() {
  const { id: projectId } = useParams<{ id: string }>();
  const navigate = useNavigate();
  const stageRef = useRef<Konva.Stage>(null);
  const { workOrder, isLoading, create, update, generatePdf } = useWorkOrder(projectId!);

  const [scale, setScale] = useState(1);
  const [canvasSize, setCanvasSize] = useState({ width: 800, height: 600 });

    if (workOrder) {
      const dd = workOrder.drawingData as { elements?: CanvasElement[] };
      if (dd.elements) setElements(dd.elements);
      setSegments(workOrder.segments);
      setPropertyNotes(workOrder.propertyNotes || '');
    }
  }, [workOrder]);

  // Resize canvas to fill container
  useEffect(() => {

      const w = window.innerWidth - 400; // minus sidebar + panel
      const h = window.innerHeight - 160;
      setCanvasSize({ width: Math.max(w, 400), height: Math.max(h, 400) });
    }
    handleResize();
    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  const pushUndo = useCallback(() => {
    setUndoStack((prev) => [...prev.slice(-20), [...elements]]);
    setRedoStack([]);
  }, [elements]);

  const undo = () => {
    if (undoStack.length === 0) return;
    setRedoStack((prev) => [...prev, [...elements]]);
    setElements(undoStack[undoStack.length - 1]);
    setUndoStack((prev) => prev.slice(0, -1));
  };

  const redo = () => {
    if (redoStack.length === 0) return;
    setUndoStack((prev) => [...prev, [...elements]]);
    setElements(redoStack[redoStack.length - 1]);
    setRedoStack((prev) => prev.slice(0, -1));
  };

  const snapToGrid = (val: number) => Math.round(val / GRID_SIZE) * GRID_SIZE;

  const handleStageClick = (e: Konva.KonvaEventObject<MouseEvent>) => {
    const stage = e.target.getStage();
    if (!stage) return;
    const pos = stage.getRelativePointerPosition();
    if (!pos) return;
    const x = snapToGrid(pos.x);
    const y = snapToGrid(pos.y);

    if (activeTool === 'house') {
      pushUndo();
      setElements((prev) => [...prev, {
        id: crypto.randomUUID(),
        type: 'house',
        x, y,
        width: GRID_SIZE * 8,
        height: GRID_SIZE * 6,

      setActiveTool('select');
    } else if (activeTool === 'fence') {
      if (!drawingFence) {
        setDrawingFence(true);
        setFencePoints([x, y]);
      } else {
        setFencePoints((prev) => [...prev, x, y]);
      }
    } else if (activeTool === 'text') {
      const textContent = prompt('Enter label:');
      if (textContent) {
        pushUndo();
        setElements((prev) => [...prev, { id: crypto.randomUUID(), type: 'text', x, y, text: textContent }]);

  };

    { tool: 'gate', icon: MapPin, label: 'Gate' },
    { tool: 'text', icon: Type, label: 'Label' },

    return <div className="flex justify-center items-center h-screen"><div className="animate-spin rounded-full h-8 w-8 border-b-2 border-brand-purple" /></div>;

                {/* Additions */}
                <div>
                  <label className="text-xs font-medium text-gray-500">Additions</label>
                  <div className="flex flex-wrap gap-1 mt-1">
                    {ADDITIONS.map((add) => {
                      const isActive = segments[selectedFenceOrder].additions.includes(add);
                      return (
                        <Badge key={add} variant={isActive ? 'default' : 'outline'} className="cursor-pointer text-xs"
                          onClick={() => {
                            const additions = isActive
                              ? segments[selectedFenceOrder].additions.filter((a) => a !== add)
                              : [...segments[selectedFenceOrder].additions, add];
                            updateSegment(selectedFenceOrder, { additions });
                          }}>
                          {add}
                        </Badge>
                      );
                    })}
                  </div>
                  <Button variant="ghost" size="sm" className="mt-1 text-xs" onClick={() => {
                    const custom = prompt('Enter custom addition:');
                    if (custom) updateSegment(selectedFenceOrder, { customAdditions: [...segments[selectedFenceOrder].customAdditions, custom] });
                  }}>+ Add Custom</Button>
                  {segments[selectedFenceOrder].customAdditions.map((ca, i) => (
                    <Badge key={i} variant="default" className="text-xs mr-1">{ca}</Badge>
                  ))}
                </div>

                <div>
                  <label className="text-xs font-medium text-gray-500">Notes</label>
                  <textarea className="w-full border rounded px-2 py-1 text-sm" rows={2} value={segments[selectedFenceOrder].notes || ''}
                    onChange={(e) => updateSegment(selectedFenceOrder, { notes: e.target.value })} />
                </div>
              </CardContent>
            </Card>
          ) : (
            <div className="text-sm text-gray-500 text-center py-8">
              Select a fence segment to edit properties, or use the toolbar to draw.
            </div>
          )}

          {/* Property Notes */}
          <Card>
            <CardHeader className="pb-2"><CardTitle className="text-sm">Property Notes</CardTitle></CardHeader>
            <CardContent>
              <textarea className="w-full border rounded px-2 py-1 text-sm" rows={3} value={propertyNotes} onChange={(e) => setPropertyNotes(e.target.value)} placeholder="General site notes..." />
            </CardContent>
          </Card>

          {/* Segments Summary */}
          <Card>
            <CardHeader className="pb-2"><CardTitle className="text-sm">Summary</CardTitle></CardHeader>
            <CardContent className="space-y-1">
              <p className="text-sm"><span className="font-medium">Segments:</span> {segments.length}</p>
              <p className="text-sm"><span className="font-medium">Total Linear Feet:</span> {totalLinearFeet.toFixed(1)}'</p>
              {segments.map((seg, i) => (
                <div key={i} className="text-xs text-gray-600 flex justify-between border-t pt-1">
                  <span>#{i + 1} {seg.fenceType}</span>
                  <span>{seg.linearFeet}' x {seg.height}'</span>
                </div>
              ))}
            </CardContent>
          </Card>
        </div>
      </div>
    </div>
  );
}
```
